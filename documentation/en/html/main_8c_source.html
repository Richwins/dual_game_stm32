65/**
66 * @brief Lit la valeur ADC 12 bits d'un canal donné.
67 * @details Réinitialise le périphérique ADC, configure le canal demandé en
68 * mode single-shot, lance la calibration, active l'ADC puis retourne
69 * la valeur convertie sur 12 bits (0–4095).
70 * @param channel Canal ADC LL à convertir (ex: LL_ADC_CHANNEL_0 pour PA0,
71 * LL_ADC_CHANNEL_1 pour PA1).
72 * @return Valeur ADC 12 bits (0 = 0 V, 4095 = 3,3 V).
73 */
74uint16_t ADC_ReadChannel(uint32_t channel) {
75 /* Reset complet du peripheral ADC */
76 LL_APB2_GRP1_ForceReset(LL_APB2_GRP1_PERIPH_ADC1);
77 LL_APB2_GRP1_ReleaseReset(LL_APB2_GRP1_PERIPH_ADC1);
78
79 /* Reconfigure l'ADC */
80 LL_ADC_SetResolution(ADC1, LL_ADC_RESOLUTION_12B);
81 LL_ADC_SetDataAlignment(ADC1, LL_ADC_DATA_ALIGN_RIGHT);
82 LL_ADC_SetLowPowerMode(ADC1, LL_ADC_LP_MODE_NONE);
83 LL_ADC_REG_SetTriggerSource(ADC1, LL_ADC_REG_TRIG_SOFTWARE);
84 LL_ADC_REG_SetContinuousMode(ADC1, LL_ADC_REG_CONV_SINGLE);
85 LL_ADC_REG_SetDMATransfer(ADC1, LL_ADC_REG_DMA_TRANSFER_NONE);
86 LL_ADC_REG_SetOverrun(ADC1, LL_ADC_REG_OVR_DATA_OVERWRITTEN);
87 LL_ADC_SetSamplingTimeCommonChannels(ADC1, LL_ADC_SAMPLINGTIME_39CYCLES_5);
88
89 /* Ajoute uniquement le canal voulu */
90 LL_ADC_REG_SetSequencerChAdd(ADC1, channel);
91
92 /* Calibration */
93 LL_ADC_StartCalibration(ADC1);
94 while (LL_ADC_IsCalibrationOnGoing(ADC1))
95 ;
96
97 /* Active */
98 LL_ADC_Enable(ADC1);
99 while (!LL_ADC_IsActiveFlag_ADRDY(ADC1))
100 ;
101
102 /* Convertit */
103 LL_ADC_ClearFlag_EOC(ADC1);
104 LL_ADC_REG_StartConversion(ADC1);
105 uint32_t timeout = 1000000;
106 while (!LL_ADC_IsActiveFlag_EOC(ADC1) && timeout--)
107 ;
108
109 return LL_ADC_REG_ReadConversionData12(ADC1);
110}

112/**
113 * @brief Fait clignoter une LED un certain nombre de fois.
114 * @param GPIOx Port GPIO de la LED (ex: GPIOB, GPIOC).
115 * @param pin Broche LL de la LED (ex: LL_GPIO_PIN_6).
116 * @param count Nombre de clignotements (chaque cycle = 500 ms).
117 */
118void Blink(GPIO_TypeDef* GPIOx, uint32_t pin, uint8_t count) {
119 for (uint8_t i = 0; i < count; i++) {
120 LL_GPIO_ResetOutputPin(GPIOx, pin);
121 LL_mDelay(250);
122 LL_GPIO_SetOutputPin(GPIOx, pin);
123 LL_mDelay(250);
124 LL_GPIO_ResetOutputPin(GPIOx, pin);
125 }
126}
