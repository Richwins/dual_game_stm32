81 /**
82 * @brief Ajoute un message horodaté au journal des événements.
83 * @param {string} msg - Message à enregistrer.
84 */
85 function log(msg) {
86 const t = new Date().toLocaleTimeString("fr-FR", {
87 hour: "2-digit",
88 minute: "2-digit",
89 second: "2-digit",
90 });
91 state.messages.unshift(`[${t}] ${msg}`);

95 /**
96 * @brief Contraint une valeur entre un minimum et un maximum.
97 * @param {number} val - Valeur à contraindre.
98 * @param {number} min - Borne inférieure.
99 * @param {number} max - Borne supérieure.
100 * @return {number} Valeur contrainte dans [min, max].
101 */
102 function clamp(val, min, max) {
103 return Math.max(min, Math.min(max, val));
104 }

106 function dist(a, b) {
107 return Math.sqrt((a.x - b.x) ** 2 + (a.y - b.y) ** 2);
108 }

110 /**
111 * @brief Retourne l'objet joueur adverse.
112 * @param {number} playerId - Identifiant du joueur courant (1 ou 2).
113 * @return {Object|undefined} L'objet joueur adverse.

115 function getOpponent(playerId) {
116 return state.players.find((p) => p.id !== playerId);

120 * @brief Remet les deux joueurs à leur état initial de manche.

147 /**
148 * @brief Lance le compte à rebours 3-2-1 avant le début du combat.
149 * @details Passe la phase en "countdown", décrémente chaque seconde,
150 * puis passe en "playing" quand le compteur atteint 0.

152 function startCountdown() {
153 state.phase = "countdown";
154 state.countdown = 3;
155 if (countdownTimer) clearInterval(countdownTimer);
156 countdownTimer = setInterval(() => {
157 state.countdown--;
158 if (state.countdown <= 0) {
159 clearInterval(countdownTimer);
160 countdownTimer = null;
161 state.phase = "playing";
162 log("COMBAT !");
163 }
164 }, 1000);

167 /**
168 * @brief Initialise et lance la manche courante.
169 * @details Remet les joueurs à zéro et démarre le compte à rebours.

178 /**
179 * @brief Termine la manche et désigne le vainqueur.
180 * @details Passe la phase en "finished", incrémente les victoires du gagnant
181 * et active le prompt de relance après 2,5 secondes.
182 * @param {number} winnerId - Identifiant du joueur vainqueur (1 ou 2).
183 */
184 function endRound(winnerId) {
185 if (countdownTimer) {
186 clearInterval(countdownTimer);
187 countdownTimer = null;
188 }
189 state.phase = "finished";
190 state.winnerId = winnerId;
191 state.readyToRestart = false;
192 const w = state.players.find((p) => p.id === winnerId);
193 if (w) {
194 w.wins++;
195 log(`${w.name} remporte la manche ${state.round} !`);
196 }
197 // Affiche le prompt "Appuie sur 1" après 2,5 secondes
198 setTimeout(() => {
199 if (state.phase === "finished") state.readyToRestart = true;
200 }, 2500);
201 }

203 /**
204 * @brief Passe à la manche suivante (incrémente le numéro et relance).

211 /**
212 * @brief Déclenche le redémarrage de manche sur pression du bouton "1".
213 * @details Ignoré si la phase n'est pas "finished" ou si readyToRestart
214 * est encore false (délai de 2,5 s non écoulé).

216 function triggerRestart() {
217 if (state.phase !== "finished") return;
218 if (!state.readyToRestart) return;

222 /**
223 * @brief Réinitialise complètement le match (manche 1, scores et victoires à zéro).
224 */
225 function resetMatch() {
226 if (countdownTimer) clearInterval(countdownTimer);
227 state.readyToRestart = false;
228 state.round = 1;
229 state.messages = [];
230 state.players.forEach((p) => {
231 p.wins = 0;
232 p.score = MAX_HP * 10;
233 });
234 startRound();
235 }

237 const HALF = ARENA_W / 2;
238
239 /**
240 * @brief Déplace un joueur d'un pas dans la direction indiquée.
241 * @details Ignoré si la phase n'est pas "playing". Le joueur est contraint
242 * dans sa demi-arène (séparée par ZONE_GAP) et oriente toujours
243 * son avion vers l'adversaire.
244 * @param {number} playerId - Identifiant du joueur (1 ou 2).
245 * @param {string} direction - Direction : "haut" | "bas" | "gauche" | "droite".
246 */
247 function movePlayer(playerId, direction) {
248 if (state.phase !== "playing") return;
249 const p = state.players.find((pl) => pl.id === playerId);
250 if (!p || p.hp <= 0) return;
251
252 // Chaque joueur est limité à sa zone — une zone neutre de ZONE_GAP sépare les deux
253 const xMin = playerId === 1 ? PLAYER_R + 2 : HALF + ZONE_GAP / 2 + PLAYER_R + 2;
254 const xMax = playerId === 1 ? HALF - ZONE_GAP / 2 - PLAYER_R - 2 : ARENA_W - PLAYER_R - 2;
255
256 switch (direction) {
257 case "haut":
258 p.y = clamp(p.y - MOVE_STEP, PLAYER_R + 2, ARENA_H - PLAYER_R - 2);
259 break;
260 case "bas":
261 p.y = clamp(p.y + MOVE_STEP, PLAYER_R + 2, ARENA_H - PLAYER_R - 2);
262 break;
263 case "gauche":
264 p.x = clamp(p.x - MOVE_STEP, xMin, xMax);
265 break;
266 case "droite":
267 p.x = clamp(p.x + MOVE_STEP, xMin, xMax);
268 break;
269 }
270 // toujours face à l'adversaire
271 const opp = getOpponent(playerId);
272 if (opp) p.facing = opp.x >= p.x ? 1 : -1;
273 }

275 const PROJ_SPEED = 12; // px par tick
276 const PROJ_TICK = 14; // ms par tick (~70 fps)
277 const PROJ_HIT_R = PLAYER_R + 10; // rayon de collision
278
279 let projCounter = 0;
280 const projIntervals = []; // pour tout nettoyer au reset
281
282 /**
283 * @brief Supprime tous les projectiles actifs et annule leurs intervalles.
284 */
285 function clearAllProjectiles() {
286 projIntervals.forEach(clearInterval);
287 projIntervals.length = 0;
288 state.projectiles.length = 0;
289 }

291 /**
292 * @brief Crée et anime un projectile tiré par l'attaquant.
293 * @details Le projectile se déplace à PROJ_SPEED px/tick. À l'impact,
294 * le score de la cible est decrementé de 1 et son hp recalculé.
295 * Si le score atteint 0, endRound() est appelé.
296 * @param {Object} atk - Objet joueur attaquant (possède id, x, y, facing, name).
297 */
298 function spawnProjectile(atk) {
299 const id = ++projCounter;
300 // Le tir part du bord du personnage, dans la direction qu'il regarde
301 const startX = atk.x + atk.facing * (PLAYER_R + 6);
302 state.projectiles.push({
303 id,
304 x: startX,
305 y: atk.y,
306 facing: atk.facing,
307 playerId: atk.id,
308 exploding: false,
309 });
310
311 const def = getOpponent(atk.id);
312
313 const interval = setInterval(() => {
314 const proj = state.projectiles.find((p) => p.id === id);
315 if (!proj) { clearInterval(interval); return; }
316
317 // Déplacement du projectile
318 proj.x += atk.facing * PROJ_SPEED;
319
320 // Collision avec l'adversaire
321 if (def && def.hp > 0 && !proj.exploding) {
322 const dx = Math.abs(proj.x - def.x);
323 const dy = Math.abs(proj.y - def.y);
324 if (dx <= PROJ_HIT_R && dy <= PROJ_HIT_R) {
325 proj.exploding = true;
326 clearInterval(interval);
327
328 def.score = Math.max(0, def.score - 1);
329 def.hp = Math.ceil(def.score / 10);
330 def.hit = true;
331 log(`${atk.name} touche ${def.name} ! Score: ${def.score}/${MAX_HP * 10} (${def.hp} LED${def.hp > 1 ? "s" : ""})`);
332 setTimeout(() => { def.hit = false; }, 450);
333 if (def.score <= 0) {
334 setTimeout(() => endRound(atk.id), 350);
335 }
336
337 // Laisser l'explosion visible brièvement puis supprimer
338 setTimeout(() => {
339 const idx = state.projectiles.findIndex((p) => p.id === id);
340 if (idx !== -1) state.projectiles.splice(idx, 1);
341 }, 320);
342 return;
343 }
344 }
345
346 // Hors de l'arène → supprimer
347 if (proj.x < -60 || proj.x > ARENA_W + 60) {
348 clearInterval(interval);
349 const idx = state.projectiles.findIndex((p) => p.id === id);
350 if (idx !== -1) state.projectiles.splice(idx, 1);
351 log(`${atk.name} attaque... raté !`);
352 }
353 }, PROJ_TICK);
354
355 projIntervals.push(interval);
356 }

358 /**
359 * @brief Déclenche l'attaque d'un joueur (tir d'un projectile).
360 * @details Ignoré si la phase n'est pas "playing" ou si le joueur est éliminé.
361 * L'animation d'attaque dure 350 ms.
362 * @param {number} attackerId - Identifiant du joueur qui tire (1 ou 2).
363 */
364 function attack(attackerId) {
365 if (state.phase !== "playing") return;
366 const atk = state.players.find((p) => p.id === attackerId);
367 if (!atk || atk.hp <= 0) return;
368
369 atk.attacking = true;
370 setTimeout(() => { atk.attacking = false; }, 350);
371
372 spawnProjectile(atk);
373 }

375 /**
376 * @brief Table de traduction des directions STM32 (anglais) vers le jeu (français).
377 * @type {Object.<string, string>}
378 */
379 const DIRECTION_MAP = {
380 UP: "haut",
381 DOWN: "bas",
382 LEFT: "gauche",
383 RIGHT: "droite",
384 };
385
386 /**
387 * @brief Point d'entrée unique pour les événements reçus depuis le STM32.
388 * @details Traite les événements de type "fire", "joystick" et "tick".
389 * Pendant la phase "finished", "fire" déclenche triggerRestart().
390 * Les directions sont traduites via DIRECTION_MAP avant d'appeler movePlayer().
391 * "tick" du joueur 2 incrémente le chronomètre de 1 seconde.
392 * @param {Object} event - Événement WebSocket désérialisé.
393 * @param {string} event.type - Type : "fire" | "joystick" | "tick".
394 * @param {number} event.playerId - Identifiant de la carte émettrice (1 ou 2).
395 * @param {string} [event.direction]- Direction joystick : "UP"|"DOWN"|"LEFT"|"RIGHT".
396 */
397 function handleHardwareEvent(event) {
398 switch (event.type) {
399 case "fire":
400 if (state.phase === "finished") {
401 triggerRestart();
402 break;
403 }
404 attack(event.playerId);
405 break;
406 case "tick":
407 if (event.playerId === 2) {
408 state.elapsedSeconds++;
409 }
410 break;
411 case "joystick": {
412 const dir = DIRECTION_MAP[event.direction] ?? event.direction;
413 movePlayer(event.playerId, dir);
414 break;
415 }
416 default:
417 log(`[STM32] Événement inconnu: ${JSON.stringify(event)}`);
418 }
419 }
